  if(id==="I"){const D=D_SiC("Kr",T_C),eta=sphEta(D,t_d*86400,35e-6);return Math.min(1,fd*(eta*0.2+GAP_F.I));}
  return 0;
}

function calcAccRF(id,T_C,t_h,sc,coolant){
  function interpST(key){
    if(T_C<=1500){
      const f=ACC_RATE[id==="Eu154"?"Eu154":id];
      if(!f)return 0;
      return Math.min(0.999,arr(f.A,f.B,T_C)*t_h);
    }
    if(T_C<=1600)return AGR_ST[1600][key]||1e-9;
    if(T_C<=1700){const lo=AGR_ST[1600][key]||1e-9,hi=AGR_ST[1700][key]||lo;return lo+(hi-lo)*(T_C-1600)/100;}
    if(T_C<=1800){const lo=AGR_ST[1700][key]||1e-9,hi=AGR_ST[1800][key]||lo;return lo+(hi-lo)*(T_C-1700)/100;}
    return Math.min(0.999,(AGR_ST[1800][key]||1e-3)*Math.pow(T_C/1800,3));
  }
  const sic_f=getSiCFail(T_C),gap_f=GAP_F[{Ag110m:"Ag",Cs134:"Cs",Sr90:"Sr",Eu154:"Eu",Kr:"Kr",I:"I"}[id]]||1e-3;
  let rf=0;
  if(id==="Ag110m")rf=interpST("Ag");
  else if(id==="Cs134")rf=interpST("Cs");
  else if(id==="Sr90")rf=interpST("Sr");
  else if(id==="Eu154")rf=interpST("Eu");
  else if(id==="Kr"){const D=D_SiC("Kr",T_C);rf=sic_f*GAP_F.Kr+(1-sic_f)*sphEta(D,t_h*3600,35e-6)*0.01;}
  else if(id==="I"){const D=D_SiC("Kr",T_C);rf=sic_f*GAP_F.I+(1-sic_f)*sphEta(D,t_h*3600,35e-6)*0.005;}
  rf=Math.max(rf,sic_f*gap_f);
  if(coolant==="air"&&!["Kr","I"].includes(id))rf=Math.min(1,rf*1.7);
  rf*={dlofc:1.0,air:1.0,sft_lop:0.15,sft_hx:0.12}[sc]||1.0;
  return Math.min(0.999,rf);
}

function getSiCFail(T_C){
  if(T_C<=1400)return SIC_95CL.base;
  if(T_C<=1600)return SIC_95CL.base+(SIC_95CL[1600]-SIC_95CL.base)*(T_C-1400)/200;
  if(T_C<=1700)return SIC_95CL[1600]+(SIC_95CL[1700]-SIC_95CL[1600])*(T_C-1600)/100;
  if(T_C<=1800)return SIC_95CL[1700]+(SIC_95CL[1800]-SIC_95CL[1700])*(T_C-1700)/100;
  return Math.min(0.15,SIC_95CL[1800]*Math.pow(T_C/1800,2));
}
